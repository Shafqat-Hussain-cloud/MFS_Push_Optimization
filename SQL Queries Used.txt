SQL Queries Used1. --Transaction Amount and Count by Day of Month--SELECT    service_name AS "Service Name",    SUM(transaction_amount) AS "Transaction Amount",    COUNT(*) AS "Transaction Count2",    EXTRACT(DAY FROM transaction_date) AS "Day of Month"FROM    transactionsGROUP BY    service_name,    EXTRACT(DAY FROM transaction_date)ORDER BY    "Day of Month";
2. --Transaction Count by Hour of Day--SELECT    service_name AS "Service Name",    COUNT(*) AS "Transaction Count",    EXTRACT(HOUR FROM transaction_date) AS "Hour"FROM    transactionsGROUP BY    service_name,    EXTRACT(HOUR FROM transaction_date)ORDER BY    EXTRACT(HOUR FROM transaction_date)
